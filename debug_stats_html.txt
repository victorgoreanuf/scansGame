<script>
(function(){
  let secs = 3475;
  const refillAmt = 53;
  const maxStam = 1860;

  const tEl = document.getElementById('stamina_timer');
  const sEl = document.getElementById('stamina_span');

  function fmt(n){ return n.toLocaleString(); }
  function mmss(total){
    const m = Math.floor(total / 60);
    const s = total % 60;
    return (m<10?'0':'')+m+':' + (s<10?'0':'')+s;
  }

  function tick(){
    tEl.textContent = '⏳ ' + mmss(secs);

    if (secs <= 0){
      // Apply +10 stamina visually (server already did at HH:00:00)
      const current = parseInt((sEl.textContent || '0').replace(/[^0-9]/g,''), 10) || 0;
      const next = Math.min(maxStam, current + refillAmt);
      if (next !== current){
        sEl.textContent = fmt(next);
      }
      secs = 3600;
    }
    secs--;
  }

  tick();
  setInterval(tick, 1000);
})();
</script>
<script>
(function(){
  const el = document.getElementById('server_time');
  if (!el) return;

  let epoch  = parseInt(el.getAttribute('data-epoch'), 10) || Math.floor(Date.now()/1000);
  const offS = parseInt(el.getAttribute('data-tzoff'), 10) || 0; // server offset (seconds from UTC)

  function pad2(n){ return n < 10 ? '0' + n : '' + n; }

  function render(){
    // Shift by server offset, then read UTC fields => server-local time
    const d = new Date((epoch + offS) * 1000);
    let h = d.getUTCHours();
    const m = d.getUTCMinutes();
    const s = d.getUTCSeconds();
    const am = h < 12;
    let h12 = h % 12; if (h12 === 0) h12 = 12;
    el.textContent = h12 + ':' + pad2(m) + ':' + pad2(s) + ' ' + (am ? 'AM' : 'PM');
  }

  render();
  setInterval(function(){ epoch++; render(); }, 1000);
})();
</script>

<script>
(function(){
  const btn   = document.getElementById('buffs_btn');
  const modal = document.getElementById('buffs_modal');
  const close = document.getElementById('buffs_close');
  const list  = document.getElementById('buffs_list');

  if (!btn || !modal || !list) return;

  // Open / close
  btn.addEventListener('click', () => { modal.classList.add('open'); modal.setAttribute('aria-hidden','false'); });
  close.addEventListener('click', () => { modal.classList.remove('open'); modal.setAttribute('aria-hidden','true'); });
  modal.addEventListener('click', (e) => {
    if (e.target === modal) { modal.classList.remove('open'); modal.setAttribute('aria-hidden','true'); }
  });
  window.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && modal.classList.contains('open')) {
      modal.classList.remove('open'); modal.setAttribute('aria-hidden','true');
    }
  });

  // Countdown: align to server time to avoid client skew
  const serverStartMs = (parseInt(document.getElementById('server_time')?.getAttribute('data-epoch') || '0', 10)) * 1000;
  const offsetMs = serverStartMs ? (serverStartMs - Date.now()) : 0; // server - client

  function nowServerSec(){ return Math.floor((Date.now() + offsetMs) / 1000); }
  function fmtLeft(secs){
    if (secs < 0) secs = 0;
    const h = Math.floor(secs/3600);
    const m = Math.floor((secs%3600)/60);
    const s = secs%60;
    return (h>0 ? (h+':'+String(m).padStart(2,'0')+':') : (m+':')) + String(s).padStart(2,'0');
  }

  function tick(){
    const now = nowServerSec();
    list.querySelectorAll('.buff-row').forEach(row => {
      const ends = parseInt(row.getAttribute('data-ends') || '0', 10);
      const left = Math.max(0, ends - now);
      const label = row.querySelector('[data-ends-text]');
      if (label) label.textContent = left > 0 ? (fmtLeft(left) + ' left') : 'expired';
    });
  }

  tick();
  setInterval(tick, 1000);
})();
</script>

            if (setNum > 0) {
                openModal(setNum, applyType);
            }
            return;

        if (!bodyEl.classList.contains('qsmodal-open')) return;

        // Click outside dialog on the dark backdrop

        if (!actionBtn) return;

        if (target === 'cancel') {

            closeModal();

        }

    async function applyQuickSet(setNum, target, actionBtn) {
        const originalText = actionBtn.textContent;
        actionBtn.disabled = true;
        actionBtn.textContent = 'Applying...';

        const params = new URLSearchParams();
        params.set('set_number', String(setNum));
        params.set('target_set', target);          // "attack" or "defense"
        params.set('apply_type', selectedApplyType); // "equipment" or "pets"

            const res = await fetch('quick_sets_apply.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: params.toString()
            });
            const raw   = await res.text();
            const lower = raw.trim().toLowerCase();

                showNotification(
                    `Quick Set #${setNum} equipped as your main ${target.toUpperCase()} set.`,
                    'success'
                );
                closeModal();
                // If you want immediate refresh to reflect new stats/gear:
                setTimeout(() => { location.reload(); }, 600);
            } else {
                showNotification(raw.slice(0, 200) || 'Failed to apply quick set.', 'error');
                actionBtn.disabled = false;
                actionBtn.textContent = originalText;
            }
        } catch (err) {
            showNotification('Server error while applying quick set.', 'error');
            actionBtn.disabled = false;
            actionBtn.textContent = originalText;
        }
    }

<script>
function alloc(stat, amount){
  const niceName = (stat === 'attack') ? 'ATTACK'
                 : (stat === 'defense') ? 'DEFENSE'
                 : (stat === 'stamina') ? 'MAX STAMINA'
                 : stat.toUpperCase();

  const msg = `Spend ${amount} stat point${amount === 1 ? '' : 's'} to add +${amount} to ${niceName}?`;
  if (!confirm(msg)) return;

  const body = `action=allocate&stat=${encodeURIComponent(stat)}&amount=${encodeURIComponent(amount)}`;
  fetch('stats_ajax.php', {
    method:'POST',
    headers:{'Content-Type':'application/x-www-form-urlencoded'},
    body
  })
  .then(async r => {
    const txt = await r.text();
    try { return { okHTTP: r.ok, json: JSON.parse(txt), raw: txt }; }
    catch { throw new Error(`Bad JSON (${r.status}): ${txt}`); }
  })
  .then(pack => {
    if (!pack.okHTTP) { alert(`HTTP ${pack.raw}`); return; }
    const res = pack.json;
    if (!res.ok) { alert(res.msg || 'Error'); return; }

    const u = res.user;
    document.getElementById('v-points').textContent  = u.STAT_POINTS;
    document.getElementById('v-attack').textContent  = u.ATTACK;
    document.getElementById('v-defense').textContent = u.DEFENSE;
    document.getElementById('v-stamina').textContent = u.MAX_STAMINA;
  })
  .catch(err => {
    console.error(err);
    alert(err.message || 'Network error');
  });
}

</script>

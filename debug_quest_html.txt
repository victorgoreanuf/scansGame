<script>
(function(){
  let secs = 357;
  const refillAmt = 53;
  const maxStam = 1860;

  const tEl = document.getElementById('stamina_timer');
  const sEl = document.getElementById('stamina_span');

  function fmt(n){ return n.toLocaleString(); }
  function mmss(total){
    const m = Math.floor(total / 60);
    const s = total % 60;
    return (m<10?'0':'')+m+':' + (s<10?'0':'')+s;
  }

  function tick(){
    tEl.textContent = '⏳ ' + mmss(secs);

    if (secs <= 0){
      // Apply +10 stamina visually (server already did at HH:00:00)
      const current = parseInt((sEl.textContent || '0').replace(/[^0-9]/g,''), 10) || 0;
      const next = Math.min(maxStam, current + refillAmt);
      if (next !== current){
        sEl.textContent = fmt(next);
      }
      secs = 3600;
    }
    secs--;
  }

  tick();
  setInterval(tick, 1000);
})();
</script>
<script>
(function(){
  const el = document.getElementById('server_time');
  if (!el) return;

  let epoch  = parseInt(el.getAttribute('data-epoch'), 10) || Math.floor(Date.now()/1000);
  const offS = parseInt(el.getAttribute('data-tzoff'), 10) || 0; // server offset (seconds from UTC)

  function pad2(n){ return n < 10 ? '0' + n : '' + n; }

  function render(){
    // Shift by server offset, then read UTC fields => server-local time
    const d = new Date((epoch + offS) * 1000);
    let h = d.getUTCHours();
    const m = d.getUTCMinutes();
    const s = d.getUTCSeconds();
    const am = h < 12;
    let h12 = h % 12; if (h12 === 0) h12 = 12;
    el.textContent = h12 + ':' + pad2(m) + ':' + pad2(s) + ' ' + (am ? 'AM' : 'PM');
  }

  render();
  setInterval(function(){ epoch++; render(); }, 1000);
})();
</script>

<script>
(function(){
  const btn   = document.getElementById('buffs_btn');
  const modal = document.getElementById('buffs_modal');
  const close = document.getElementById('buffs_close');
  const list  = document.getElementById('buffs_list');

  if (!btn || !modal || !list) return;

  // Open / close
  btn.addEventListener('click', () => { modal.classList.add('open'); modal.setAttribute('aria-hidden','false'); });
  close.addEventListener('click', () => { modal.classList.remove('open'); modal.setAttribute('aria-hidden','true'); });
  modal.addEventListener('click', (e) => {
    if (e.target === modal) { modal.classList.remove('open'); modal.setAttribute('aria-hidden','true'); }
  });
  window.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && modal.classList.contains('open')) {
      modal.classList.remove('open'); modal.setAttribute('aria-hidden','true');
    }
  });

  // Countdown: align to server time to avoid client skew
  const serverStartMs = (parseInt(document.getElementById('server_time')?.getAttribute('data-epoch') || '0', 10)) * 1000;
  const offsetMs = serverStartMs ? (serverStartMs - Date.now()) : 0; // server - client

  function nowServerSec(){ return Math.floor((Date.now() + offsetMs) / 1000); }
  function fmtLeft(secs){
    if (secs < 0) secs = 0;
    const h = Math.floor(secs/3600);
    const m = Math.floor((secs%3600)/60);
    const s = secs%60;
    return (h>0 ? (h+':'+String(m).padStart(2,'0')+':') : (m+':')) + String(s).padStart(2,'0');
  }

  function tick(){
    const now = nowServerSec();
    list.querySelectorAll('.buff-row').forEach(row => {
      const ends = parseInt(row.getAttribute('data-ends') || '0', 10);
      const left = Math.max(0, ends - now);
      const label = row.querySelector('[data-ends-text]');
      if (label) label.textContent = left > 0 ? (fmtLeft(left) + ' left') : 'expired';
    });
  }

  tick();
  setInterval(tick, 1000);
})();
</script>

        try {
            const endpoint = (parseInt(itemId, 10) === 108) ? 'user_heal_potion.php' : 'use_item.php';

            const res = await fetch(endpoint, {
                method: 'POST',
                headers: {'Content-Type': 'application/x-www-form-urlencoded'},
                body: params.toString()
            });

            const raw = await res.text();
            const lower = raw.trim().toLowerCase();

            if (lower.includes('success')) {
                showNotification('Used ' + itemNm + '!', 'success');

                // update qty inside drawer
                let leftNow = parseInt(qtyLeftSpan.textContent,10) || 0;
                leftNow = leftNow - useQty;
                if (leftNow < 0) leftNow = 0;
                qtyLeftSpan.textContent = leftNow;

                // update btn / input max
                btn.setAttribute('data-max', leftNow);
                if (qtyInput) {
                    qtyInput.setAttribute('max', leftNow);
                    if (leftNow === 0) qtyInput.value = 0;
                }

                // update the bubble on trigger
                const totalBubble = document.getElementById('drawerTotalQty');
                if (totalBubble) {
                    let totalAll = 0;
                    document.querySelectorAll('.potion-qty-left').forEach(sp => {
                        totalAll += (parseInt(sp.textContent,10) || 0);
                    });
                    totalBubble.textContent = totalAll;
                }

                // if out of stock disable button
                if (leftNow <= 0) {
                    btn.disabled = true;
                    btn.textContent = 'None';
                } else {
                    btn.disabled = false;
                    btn.textContent = 'Use';
                }

                // optional refresh to sync stamina/HP/etc
                location.reload();
            } else {
                showNotification(raw.slice(0,200) || 'Failed to use item.', 'error');
                btn.disabled = false;
                btn.textContent = 'Use';
            }
        } catch (err) {
            showNotification('Server error.', 'error');
            btn.disabled = false;
            btn.textContent = 'Use';
        }
    });
})();


(function(){
    let selectedSetNum   = null;
    let selectedApplyType = 'equipments';


    const bodyEl        = document.body;
    const modalBackdrop = document.getElementById('qsModalBackdrop');
    const modal         = document.getElementById('qsModal');
    const setSpan       = document.getElementById('qsModalSetNum');

        function openModal(setNum, applyType) {
        selectedSetNum    = setNum;
        selectedApplyType = (applyType === 'pets') ? 'pets' : 'equipments';

        if (setSpan) {
            // Example: "#3 – Gear" or "#3 – Pets"
            const labelSuffix = (selectedApplyType === 'pets') ? ' – Pets' : ' – Gear';
            setSpan.textContent = '#' + setNum + labelSuffix;
        }

        bodyEl.classList.add('qsmodal-open');
        if (modal)  modal.setAttribute('aria-hidden', 'false');
    }


    function closeModal() {
        selectedSetNum = null;
        bodyEl.classList.remove('qsmodal-open');
        if (modal)  modal.setAttribute('aria-hidden', 'true');
    }


    document.addEventListener('click', function(e){
        // Click on a quick set button inside the drawer
        const setBtn = e.target.closest('.qs-set-btn');
        if (setBtn) {
            const setNum    = parseInt(setBtn.getAttribute('data-set-number') || '0', 10);
            const applyType = (setBtn.getAttribute('data-apply-type') || 'equipments').toLowerCase();

            if (setNum > 0) {
                openModal(setNum, applyType);
            }
            return;
        }

        if (!bodyEl.classList.contains('qsmodal-open')) return;

        // Click outside dialog on the dark backdrop
        if (e.target === modalBackdrop) {
            closeModal();
            return;
        }

        const actionBtn = e.target.closest('.qs-modal-btn');
        if (!actionBtn) return;

        const target = actionBtn.getAttribute('data-target');
        if (target === 'cancel') {
            closeModal();
            return;
        }

        if (!selectedSetNum || (target !== 'attack' && target !== 'defense')) {
            closeModal();
            return;
        }

        applyQuickSet(selectedSetNum, target, actionBtn);
    });

    async function applyQuickSet(setNum, target, actionBtn) {
        const originalText = actionBtn.textContent;
        actionBtn.disabled = true;
        actionBtn.textContent = 'Applying...';

        const params = new URLSearchParams();
        params.set('set_number', String(setNum));
        params.set('target_set', target);          // "attack" or "defense"
        params.set('apply_type', selectedApplyType); // "equipment" or "pets"

        try {
            const res = await fetch('quick_sets_apply.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: params.toString()
            });
            const raw   = await res.text();
            const lower = raw.trim().toLowerCase();

            if (lower.includes('success')) {
                showNotification(
                    `Quick Set #${setNum} equipped as your main ${target.toUpperCase()} set.`,
                    'success'
                );
                closeModal();
                // If you want immediate refresh to reflect new stats/gear:
                setTimeout(() => { location.reload(); }, 600);
            } else {
                showNotification(raw.slice(0, 200) || 'Failed to apply quick set.', 'error');
                actionBtn.disabled = false;
                actionBtn.textContent = originalText;
            }
        } catch (err) {
            showNotification('Server error while applying quick set.', 'error');
            actionBtn.disabled = false;
            actionBtn.textContent = originalText;
        }
    }

    // ESC closes modal
    window.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && bodyEl.classList.contains('qsmodal-open')) {
            closeModal();
        }
    });
})();

let questCooldownReloadQueued = false;

function formatQuestCooldown(totalSeconds) {
    totalSeconds = Math.max(0, Math.floor(totalSeconds));
    const days = Math.floor(totalSeconds / 86400);
    const hours = Math.floor((totalSeconds % 86400) / 3600);
    const minutes = Math.floor((totalSeconds % 3600) / 60);
    const seconds = totalSeconds % 60;
    const pad = n => String(n).padStart(2, '0');

    if (days > 0) {
        return `${days}d ${pad(hours)}:${pad(minutes)}:${pad(seconds)}`;
    }
    return `${pad(hours)}:${pad(minutes)}:${pad(seconds)}`;
}

function startQuestCooldowns() {
    const nodes = Array.from(document.querySelectorAll('[data-cooldown-ts]'));
    if (!nodes.length) return;
    let timer = null;

    function tick() {
        const now = Math.floor(Date.now() / 1000);
        let hasFutureCooldown = false;

        nodes.forEach(node => {
            const unlockTs = parseInt(node.getAttribute('data-cooldown-ts') || '0', 10);
            if (!unlockTs) return;

            const remaining = unlockTs - now;
            if (remaining <= 0) {
                node.textContent = 'Ready now';
                if (!questCooldownReloadQueued) {
                    questCooldownReloadQueued = true;
                    setTimeout(() => window.location.reload(), 1200);
                }
                return;
            }

            hasFutureCooldown = true;
            node.textContent = formatQuestCooldown(remaining);
        });

        if (!hasFutureCooldown) {
            clearInterval(timer);
        }
    }

    tick();
    timer = setInterval(tick, 1000);
}

document.addEventListener('DOMContentLoaded', startQuestCooldowns);

function acceptQuest(qid, btn) {
    if (!confirm("Accept this quest from the Adventurer's Guild?")) return;
    btn.disabled = true;

    fetch('/adventurers_accept_quest.php', {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: new URLSearchParams({ quest_id: qid })
    })
    .then(r => r.json())
    .then(data => {
        if (data && data.status === 'ok') {
            alert(data.message || 'Quest accepted!');
            location.reload();
        } else {
            alert((data && data.message) ? data.message : 'Could not accept quest.');
            btn.disabled = false;
        }
    })
    .catch(() => {
        alert('Server error while accepting quest.');
        btn.disabled = false;
    });
}

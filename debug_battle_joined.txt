<script>
(function(){
  let secs = 3438;
  const refillAmt = 53;
  const maxStam = 1860;

  const tEl = document.getElementById('stamina_timer');
  const sEl = document.getElementById('stamina_span');

  function fmt(n){ return n.toLocaleString(); }
  function mmss(total){
    const m = Math.floor(total / 60);
    const s = total % 60;
    return (m<10?'0':'')+m+':' + (s<10?'0':'')+s;
  }

  function tick(){
    tEl.textContent = '⏳ ' + mmss(secs);

    if (secs <= 0){
      // Apply +10 stamina visually (server already did at HH:00:00)
      const current = parseInt((sEl.textContent || '0').replace(/[^0-9]/g,''), 10) || 0;
      const next = Math.min(maxStam, current + refillAmt);
      if (next !== current){
        sEl.textContent = fmt(next);
      }
      secs = 3600;
    }
    secs--;
  }

  tick();
  setInterval(tick, 1000);
})();
</script>
<script>
(function(){
  const el = document.getElementById('server_time');
  if (!el) return;

  let epoch  = parseInt(el.getAttribute('data-epoch'), 10) || Math.floor(Date.now()/1000);
  const offS = parseInt(el.getAttribute('data-tzoff'), 10) || 0; // server offset (seconds from UTC)

  function pad2(n){ return n < 10 ? '0' + n : '' + n; }

  function render(){
    // Shift by server offset, then read UTC fields => server-local time
    const d = new Date((epoch + offS) * 1000);
    let h = d.getUTCHours();
    const m = d.getUTCMinutes();
    const s = d.getUTCSeconds();
    const am = h < 12;
    let h12 = h % 12; if (h12 === 0) h12 = 12;
    el.textContent = h12 + ':' + pad2(m) + ':' + pad2(s) + ' ' + (am ? 'AM' : 'PM');
  }

  render();
  setInterval(function(){ epoch++; render(); }, 1000);
})();
</script>

<script>
(function(){
  const btn   = document.getElementById('buffs_btn');
  const modal = document.getElementById('buffs_modal');
  const close = document.getElementById('buffs_close');
  const list  = document.getElementById('buffs_list');

  if (!btn || !modal || !list) return;

  // Open / close
  btn.addEventListener('click', () => { modal.classList.add('open'); modal.setAttribute('aria-hidden','false'); });
  close.addEventListener('click', () => { modal.classList.remove('open'); modal.setAttribute('aria-hidden','true'); });
  modal.addEventListener('click', (e) => {
    if (e.target === modal) { modal.classList.remove('open'); modal.setAttribute('aria-hidden','true'); }
  });
  window.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && modal.classList.contains('open')) {
      modal.classList.remove('open'); modal.setAttribute('aria-hidden','true');
    }
  });

  // Countdown: align to server time to avoid client skew
  const serverStartMs = (parseInt(document.getElementById('server_time')?.getAttribute('data-epoch') || '0', 10)) * 1000;
  const offsetMs = serverStartMs ? (serverStartMs - Date.now()) : 0; // server - client

  function nowServerSec(){ return Math.floor((Date.now() + offsetMs) / 1000); }
  function fmtLeft(secs){
    if (secs < 0) secs = 0;
    const h = Math.floor(secs/3600);
    const m = Math.floor((secs%3600)/60);
    const s = secs%60;
    return (h>0 ? (h+':'+String(m).padStart(2,'0')+':') : (m+':')) + String(s).padStart(2,'0');
  }

  function tick(){
    const now = nowServerSec();
    list.querySelectorAll('.buff-row').forEach(row => {
      const ends = parseInt(row.getAttribute('data-ends') || '0', 10);
      const left = Math.max(0, ends - now);
      const label = row.querySelector('[data-ends-text]');
      if (label) label.textContent = left > 0 ? (fmtLeft(left) + ' left') : 'expired';
    });
  }

  tick();
  setInterval(tick, 1000);
})();
</script>

  function switchRoom(next){
  if (next === room) return;
  room = next;
  saveRoom(room);
  setActiveTab();
  lastHTML = '';

  // Handle access
  if (room === 'guild' && !guildId){
    stopPolling();
    elLog.innerHTML = `
      <div class="msg"><div class="content">
        <span class="u">System</span>
        <span class="t">now</span>
        <div class="text">You are not in a guild yet. Join or create a guild to use Guild Chat.</div>
      </div></div>`;
    if (elForm){ elInput.disabled = true; elSend.disabled = true; }
    return;
  }
  if (room === 'council' && !hasCouncil){
    stopPolling();
    elLog.innerHTML = `
      <div class="msg"><div class="content">
        <span class="u">System</span>
        <span class="t">now</span>
        <div class="text">Council Chat is restricted to guild leaders and vice leaders.</div>
      </div></div>`;
    if (elForm){ elInput.disabled = true; elSend.disabled = true; }
    return;
  }

  if (elForm){ elInput.disabled = false; elSend.disabled = false; }
  fetchAndRender(true);
  if (isOpen) startPolling();
}


  function startPolling(){
    stopPolling();
    if (room === 'guild' && !guildId) return; // nothing to poll
    if (room === 'council' && !hasCouncil) return; 
    pollTimer = setInterval(fetchAndRender, 1500);
  }
  function stopPolling(){
    if (pollTimer){ clearInterval(pollTimer); pollTimer = null; }
  }
  document.addEventListener('visibilitychange', () => {
    if (document.hidden || !isOpen) { stopPolling(); }
    else { startPolling(); fetchAndRender(); }
  });

  async function fetchAndRender(first=false){

    if (room === 'guild'   && !guildId)   return;
    if (room === 'council' && !hasCouncil) return; // ← add

    const wasNearBottom = (logEl.scrollHeight - logEl.scrollTop - logEl.clientHeight) < 120;
    const { update } = currentEndpoints();
    const r = await fetch(update, { method:'POST', cache:'no-store' });
    if (!r.ok) return;
    const html = await r.text();
    if (html !== lastHTML){
      lastHTML = html;
      elLog.innerHTML = html;
      if (first || wasNearBottom) logEl.scrollTop = logEl.scrollHeight;
    }
  }catch(e){ /* silent */ }
}

const replyTo = document.getElementById('gcw_reply_to')?.value || '0';

const body = 'message=' + encodeURIComponent(text)

const r = await fetch(send, {
  method:'POST',
  headers:{'Content-Type':'application/x-www-form-urlencoded'},
  body
});

        const data = await r.json().catch(()=>null);
        if (data && (''+data.status).trim() === 'success'){
          elInput.value = '';
const rt = document.getElementById('gcw_reply_to');
if (rt) rt.value = '0';
await fetchAndRender();     // <-- make it await

        }else{

        }

                    });

(function(){
    let selectedSetNum   = null;
    let selectedApplyType = 'equipments';


    const bodyEl        = document.body;
    const modalBackdrop = document.getElementById('qsModalBackdrop');
    const modal         = document.getElementById('qsModal');
    const setSpan       = document.getElementById('qsModalSetNum');

        function openModal(setNum, applyType) {
        selectedSetNum    = setNum;
        selectedApplyType = (applyType === 'pets') ? 'pets' : 'equipments';

        if (setSpan) {
            // Example: "#3 – Gear" or "#3 – Pets"
            const labelSuffix = (selectedApplyType === 'pets') ? ' – Pets' : ' – Gear';
            setSpan.textContent = '#' + setNum + labelSuffix;
        }

        bodyEl.classList.add('qsmodal-open');
        if (modal)  modal.setAttribute('aria-hidden', 'false');
    }


    function closeModal() {
        selectedSetNum = null;
        bodyEl.classList.remove('qsmodal-open');
        if (modal)  modal.setAttribute('aria-hidden', 'true');
    }


    document.addEventListener('click', function(e){
        // Click on a quick set button inside the drawer
        const setBtn = e.target.closest('.qs-set-btn');
        if (setBtn) {
            const setNum    = parseInt(setBtn.getAttribute('data-set-number') || '0', 10);
            const applyType = (setBtn.getAttribute('data-apply-type') || 'equipments').toLowerCase();

            if (setNum > 0) {
                openModal(setNum, applyType);
            }
            return;
        }

        if (!bodyEl.classList.contains('qsmodal-open')) return;

        // Click outside dialog on the dark backdrop
        if (e.target === modalBackdrop) {
            closeModal();
            return;
        }

        const actionBtn = e.target.closest('.qs-modal-btn');
        if (!actionBtn) return;

        const target = actionBtn.getAttribute('data-target');
        if (target === 'cancel') {
            closeModal();
            return;
        }

        if (!selectedSetNum || (target !== 'attack' && target !== 'defense')) {
            closeModal();
            return;
        }

        applyQuickSet(selectedSetNum, target, actionBtn);
    });

    async function applyQuickSet(setNum, target, actionBtn) {
        const originalText = actionBtn.textContent;
        actionBtn.disabled = true;
        actionBtn.textContent = 'Applying...';

        const params = new URLSearchParams();
        params.set('set_number', String(setNum));
        params.set('target_set', target);          // "attack" or "defense"
        params.set('apply_type', selectedApplyType); // "equipment" or "pets"

        try {
            const res = await fetch('quick_sets_apply.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: params.toString()
            });
            const raw   = await res.text();
            const lower = raw.trim().toLowerCase();

            if (lower.includes('success')) {
                showNotification(
                    `Quick Set #${setNum} equipped as your main ${target.toUpperCase()} set.`,
                    'success'
                );
                closeModal();
                // If you want immediate refresh to reflect new stats/gear:
                setTimeout(() => { location.reload(); }, 600);
            } else {
                showNotification(raw.slice(0, 200) || 'Failed to apply quick set.', 'error');
                actionBtn.disabled = false;
                actionBtn.textContent = originalText;
            }
        } catch (err) {
            showNotification('Server error while applying quick set.', 'error');
            actionBtn.disabled = false;
            actionBtn.textContent = originalText;
        }
    }

    // ESC closes modal
    window.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && bodyEl.classList.contains('qsmodal-open')) {
            closeModal();
        }
    });
})();

function createSparks(n=12){
  // Random short shards radiating from center
  let html = '';
  for (let i=0;i<n;i++){
    const ang = Math.random()*360;
    const dist = 20 + Math.random()*40; // vmin-ish
    const x = 50 + Math.cos(ang*Math.PI/180)*dist;
    const y = 50 + Math.sin(ang*Math.PI/180)*dist;
    const rot = Math.random()*360;
    const w = 4 + Math.floor(Math.random()*9);
    const top = `${y}vmin`, left = `${x}vmin`;
    html += `<i style="top:${top};left:${left};transform:rotate(${rot}deg);width:${w}px;"></i>`;
  }
  return html;
}

    .replace(/&/g, '&amp;')

}

function buildDamageDebugMessage(raw, data, res) {
  const parts = [];
  const msg = (data && typeof data.message === 'string') ? data.message.trim() : '';
  if (msg) parts.push(msg);
  if (data && typeof data.req_id === 'string' && data.req_id.trim()) {
    parts.push(`req_id: ${data.req_id.trim()}`);
  }
  const rawText = typeof raw === 'string' ? raw.trim() : '';
  if (rawText) {
    const normalizedMsg = msg.trim();
    if (!normalizedMsg || rawText !== normalizedMsg) {
      parts.push(rawText);
    }
  }
  if (!parts.length && res) {
    parts.push(`HTTP ${res.status}`);
  }
  return parts.join("\n\n");
}

    'error'
  );
}

       </span>`;
  } else {
    cont.innerHTML = '';
  }
}

function computeReqExp(level) {
  return Math.round(100 + Math.pow(2.5 * (level + 1), 2));
}
function addExpUI(amount) {
  const expTop  = document.querySelector('.gtb-exp-top span:last-child');
  const expFill = document.querySelector('.gtb-exp-fill');
  const lvlEl   = document.querySelector('.gtb-level');
  if (!expTop || !expFill || !lvlEl) return;
  let level = parseInt(lvlEl.textContent.replace(/[^0-9]/g, ''), 10) || 0;
  let [current, required] = expTop.textContent.split('/').map(s => parseInt(s.replace(/[^0-9]/g, ''), 10));
  if (!isFinite(current) || !isFinite(required)) return;
  current += amount;
  required = computeReqExp(level);
  while (current >= required) {
    current -= required;
    level += 1;
    required = computeReqExp(level);
  }
  lvlEl.textContent = 'LV ' + level;
  expTop.textContent = `${new Intl.NumberFormat().format(current)} / ${new Intl.NumberFormat().format(required)}`;
  expFill.style.width = Math.min(100, (current / required) * 100) + '%';
}

/* Countdown for timed heal (HH:MM:SS) */
function startHealCountdown(spanId, btnId, readyAtMs, nowMs){
  const span = document.getElementById(spanId);
  const btn  = document.getElementById(btnId);
  if (!span || !btn) return;

  let serverNow = nowMs || Date.now();
  const tw = n => String(n).padStart(2,'0');
  const fmt = s => `${tw(Math.floor(s/3600))}:${tw(Math.floor((s%3600)/60))}:${tw(s%60)}`;

  function tick(){
    const remaining = Math.max(0, Math.floor((readyAtMs - serverNow)/1000));
    if (remaining <= 0){
      span.textContent = '00:00:00';
      btn.disabled = false;
      btn.title = 'You can heal now';
      return;
    }
    span.textContent = fmt(remaining);
    serverNow += 1000;
    setTimeout(tick, 1000);
  }
  tick();
}

function fmtHMS(totalSeconds){
  totalSeconds = Math.max(0, totalSeconds|0);
  const h = Math.floor(totalSeconds / 3600);
  const m = Math.floor((totalSeconds % 3600) / 60);
  const s = totalSeconds % 60;
  const pad = n => String(n).padStart(2,'0');
  return `${pad(h)}:${pad(m)}:${pad(s)}`;
}

function startNoDamageCountdown(elId, expiresMs){
  const el = document.getElementById(elId);
  if (!el || !expiresMs) return;

  function tick(){
    const now = Date.now();
    const leftSec = Math.floor((expiresMs - now) / 1000);

    if (leftSec <= 0){
      el.textContent = "00:00:00";
      // Optional: make it obvious it has expired
      el.insertAdjacentText('afterend', ' (expired)');
      clearInterval(tmr);
      return;
    }
    el.textContent = fmtHMS(leftSec);
  }

  tick();
  const tmr = setInterval(tick, 1000);
}
